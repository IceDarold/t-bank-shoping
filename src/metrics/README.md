```python
import numpy as np
from sklearn.metrics import fbeta_score
from .base import MetricInterface

class FBetaMetric(MetricInterface):
    """
    –í—ã—á–∏—Å–ª—è–µ—Ç F-beta –º–µ—Ä—É.
    
    –≠—Ç–æ –æ–±–æ–±—â–µ–Ω–Ω–∞—è F1-–º–µ—Ä–∞, –≥–¥–µ `beta` –∫–æ–Ω—Ç—Ä–æ–ª–∏—Ä—É–µ—Ç –±–∞–ª–∞–Ω—Å
    –º–µ–∂–¥—É precision –∏ recall.
    - beta < 1: –±–æ–ª—å—à–µ –≤–µ—Å–∞ —É precision.
    - beta > 1: –±–æ–ª—å—à–µ –≤–µ—Å–∞ —É recall.
    """
    def __init__(self, beta: float, threshold: float = 0.5, average: str = 'binary'):
        if beta <= 0:
            raise ValueError("–ü–∞—Ä–∞–º–µ—Ç—Ä 'beta' –¥–æ–ª–∂–µ–Ω –±—ã—Ç—å –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω—ã–º.")
        self.beta = beta
        self.threshold = threshold
        self.average = average

    def __call__(self, y_true: np.ndarray, y_pred: np.ndarray) -> float:
        y_pred_class = (y_pred > self.threshold).astype(int)
        return fbeta_score(
            y_true, 
            y_pred_class, 
            beta=self.beta, 
            average=self.average,
            zero_division=0
        )
```
